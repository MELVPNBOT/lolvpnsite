const CONFIG = { owner: 'MELVPNBOT', repo: 'lolvpnsite', path: 'code.json', token: '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', rawUrl: 'https://raw.githubusercontent.com/MELVPNBOT/lolvpnsite/refs/heads/main/code.json' }; let captchaAnswer; function generateCaptcha() { const a = Math.floor(Math.random() * 9) + 1; const b = Math.floor(Math.random() * 9) + 1; captchaAnswer = a + b; document.getElementById('question').innerText = `${a} + ${b}`; } generateCaptcha(); async function handleVerify() { const userCaptcha = document.getElementById('captcha-input').value; const userCode = document.getElementById('user-code').value; const btn = document.getElementById('main-btn'); if (parseInt(userCaptcha) !== captchaAnswer) { alert("Капча неверна!"); generateCaptcha(); return; } btn.disabled = true; btn.innerText = "Проверка кода..."; try { // 1. Проверка через RAW (кеш-брейкер добавлен для актуальности данных) const rawRes = await fetch(CONFIG.rawUrl + '?t=' + Date.now()); const codes = await rawRes.json(); const index = codes.findIndex(c => c.code === userCode); if (index === -1) { alert("Такого кода не существует или он был удален!"); } else { btn.innerText = "Обновление базы..."; // 2. Получаем актуальный SHA через API const apiLookup = await fetch(`https://api.github.com/repos/${CONFIG.owner}/${CONFIG.repo}/contents/${CONFIG.path}`, { headers: { 'Authorization': `token ${CONFIG.token}` } }); const fileData = await apiLookup.json(); // Модифицируем массив кодов if (codes[index].activations > 1) { codes[index].activations--; } else { codes.splice(index, 1); } // 3. Пушим изменения в репозиторий const updateRes = await fetch(`https://api.github.com/repos/${CONFIG.owner}/${CONFIG.repo}/contents/${CONFIG.path}`, { method: 'PUT', headers: { 'Authorization': `token ${CONFIG.token}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ message: `Code ${userCode} activated via web`, content: btoa(JSON.stringify(codes, null, 2)), sha: fileData.sha }) }); if (updateRes.ok) { // УСПЕХ: Переход на твою саб-ссылку window.location.href = 'https://lolvpnsub.vercel.app/'; return; } else { alert("Ошибка записи в GitHub. Проверьте права токена."); } } } catch (err) { alert("Техническая ошибка. Возможно, токен недействителен."); console.error(err); } btn.disabled = false; btn.innerText = "Активировать доступ"; generateCaptcha(); }
